# OpenClaw SDR Team v1 (Core) — Cron Schedules
# 4 agents: Prospector, Email SDR, Phone SDR, Sequence Manager
# Tools: Smartlead + Airtop AI + Retell AI (no SMS)
# All times in America/New_York (ET)

# Prospector — 6 AM ET daily (pull + score + verify leads via Airtop AI + Apollo)
TZ=America/New_York
0 6 * * 1-5 cd ~/openclaw-workspace && openclaw run agents/prospector "Pull today's leads via Airtop AI (LinkedIn, Facebook, Instagram) + Apollo. Score A/B/C, verify emails, output to leads/today.json"

# Email SDR — 8 AM ET daily (first-touch emails via Smartlead)
0 8 * * 1-5 cd ~/openclaw-workspace && openclaw run agents/email-sdr "Read leads/today.json, research prospects, generate personalized first-touch emails with A/B subject lines, push to Smartlead"

# Email SDR — 10 AM ET daily (follow-up batch via Smartlead)
0 10 * * 1-5 cd ~/openclaw-workspace && openclaw run agents/email-sdr "Send follow-up emails for Day 2, Day 4, Day 7 sequences via Smartlead"

# Phone SDR — 10 AM to 12 PM ET daily (outbound calls to A-leads via Retell AI)
0 10 * * 1-5 cd ~/openclaw-workspace && openclaw run agents/phone-sdr "Call A-leads from yesterday via Retell AI. Prioritize warm leads who opened emails. Log all outcomes."

# Sequence Manager — Every 2 hours during business hours (orchestration)
0 8,10,12,14,16 * * 1-5 cd ~/openclaw-workspace && openclaw run agents/sequence-manager "Deduplicate, update CRM, check for hot leads, monitor Smartlead webhooks for speed-to-lead triggers"

# Sequence Manager — 5 PM ET daily (daily report)
0 17 * * 1-5 cd ~/openclaw-workspace && openclaw run agents/sequence-manager "Generate daily SDR report. Post to Slack #sdr. Check for hot leads to hand off."

# Sequence Manager — Friday 6 PM ET (weekly self-improvement)
0 18 * * 5 cd ~/openclaw-workspace && openclaw run agents/sequence-manager "Weekly self-improvement: analyze A/B results, update SOUL.md files, log improvements, report to Co-Founder"
